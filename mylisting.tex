\newcommand{\mylisting}[2][]{
	\bgroup
	\renewcommand*{\lstlistingname}{\tt\small Code}
	\renewcommand{\thelstlisting}{\arabic{lstlisting}:\hspace{-.3em}}
	\lstinputlisting[caption={\tt\small\detokenize{#2}},#1]{#2}
	\ifluatex
		\vspace{1\zw}
		\else
			\ifxetex
				\vspace{1em}
			\else
			\vspace{1em}
		\fi
	\fi
	\egroup
}

\lstnewenvironment{result}[2][]{
	\lstset{
		language=,
		numbers=none,
		nolol,
		title={\tt\small #2},
		#1
	}
}{
	\ifluatex
		\vspace{1\zw}
		\else
			\ifxetex
				\vspace{11pt}
			\else
			\vspace{1em}
		\fi
	\fi
}
